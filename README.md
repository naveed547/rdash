# rdash
The lodash extension

## To use
    const { uniq } = require('./rdash')

    const result = uniq([1,1,2,2,3,4,5])

    console.log(result)

## sort()
    const { sort } = require('./rdash')

    const result = sort([10,8,2,1,7,5])

    console.log(result)

## uniq()
    const { uniq } = require('./rdash')

    const result = uniq([1,1,2,2,3,4,5])

    console.log(result)
