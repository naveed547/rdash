# rdash
## The lodash extension

## To use
    const { uniq } = require('./rdash')

    var result = uniq([1,1,2,2,3,4,5])

    console.log(result)

## sort(<Array>)
    const { sort } = require('./rdash')

    var result = sort([10,8,2,1,7,5])

    console.log(result)

## uniq(<Array>)
    const { uniq } = require('./rdash')

    var result = uniq([1,1,2,2,3,4,5])

    console.log(result)
